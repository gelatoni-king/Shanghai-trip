        renderHome();

    function renderHome() {
        const container = document.getElementById('day-selector');
        container.innerHTML = "";

        travelData.forEach(day => {
            const card = document.createElement('div');
            card.className = 'card';
            card.innerHTML = `
                <div style="display:flex; justify-content:space-between; align-items:center;" onclick="showDay(${day.day})">
                    <div>
                        <span style="font-size:0.8rem; font-weight:700; color:var(--accent-peach)">DAY ${day.day}</span>
                        <h2 style="margin:5px 0; font-size:1.1rem;">${day.theme}</h2>
                        <small>${day.date}</small>
                    </div>
                    <div style="font-size:2rem;">${day.mood}</div>
                </div>
            `;
            container.appendChild(card);
        });
    }

    function showDay(dayNum) {
        const day = travelData.find(d => d.day === dayNum);
        const header = document.getElementById('day-header');
        const timeline = document.getElementById('day-timeline');

        header.innerHTML = `
            <div class="hero" style="background:${day.color}">
                <button onclick="switchView('view-home')" style="border:none; background:none; font-size:1.5rem; float:left;">🔙</button>
                <h1>Day ${day.day}</h1>
                <p>${day.date} · ${day.theme}</p>
            </div>
        `;

        timeline.innerHTML = day.events.map(e => `
            <div class="t-item" onclick="this.classList.toggle('open')">
                <div class="t-dot"></div>
                <div class="t-time">${e.time}</div>
                <div class="t-card">
                    <div style="font-weight:700;">${e.type} ${e.title}</div>
                    <div class="t-detail">
                        ${e.desc}
                        <br>
                        <a href="${e.gmaps}" target="_blank" class="btn-gmaps">📍 打開 Google Maps</a>
                    </div>
                </div>
            </div>
        `).join('');

        switchView('view-day');
    }

    function renderFood() {
        const list = document.getElementById('food-list');
        list.innerHTML = "";

        foodRecs.forEach(f => {
            list.innerHTML += `

                    <div style="font-size:2rem; margin-bottom:10px;">${f.icon}</div>

        });
    }

    function renderMap() {
        const list = document.getElementById('map-list');
        list.innerHTML = "";

        spots.forEach(s => {
            const query = encodeURIComponent(s);
            list.innerHTML += `
                <div class="card" style="display:flex; justify-content:space-between; align-items:center;">
                    <span>${s}</span>
                    <a href="https://maps.google.com/?q=${query}" target="_blank" class="btn-gmaps" style="margin-top:0">導航</a>
                </div>
            `;
        });
    }

    function switchView(viewId, btn) {
        document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
        document.getElementById(viewId).classList.add('active');

        if (btn) {
            document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
            btn.classList.add('active');
        }

        window.scrollTo(0, 0);
    }

    function toggleDarkMode() {
        const body = document.body;
        if (body.getAttribute('data-theme') === 'dark') {
            body.removeAttribute('data-theme');
        } else {
            body.setAttribute('data-theme', 'dark');
        }
    }

    function updateProgress(val) {
        document.getElementById('main-progress').style.width = val + '%';
        document.getElementById('progress-text').innerText = val;
    }
